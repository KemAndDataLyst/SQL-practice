-- counts the amount of tracks per genre and then returns the top genres with the most tracks
SELECT GenreId, COUNT(*) AS count 
FROM Track
GROUP BY GenreId 
ORDER BY count DESC 
LIMIT 5 

-- returns the total sum per country and is sorted by least to highest
SELECT BillingCountry, SUM(Total) AS sum_per_country
FROM Invoice 
GROUP BY BillingCountry 
ORDER BY BillingCountry

--returns the avg track length per genre and is sorted by Genre
SELECT GenreID, AVG(Milliseconds) AS avg_track_length
FROM Track 
GROUP BY GenreID
ORDER BY GenreID

--returns song names excluding genres 1, 2, and 3
SELECT GenreId, Name 
FROM Track 
WHERE GenreId NOT IN (1,2,3)

--returns employees hired between the specified dates
SELECT FirstName, LastName, HireDate 
FROM Employee 
WHERE HireDate BETWEEN '2002-01-01' AND '2004-12-31'


--returns the genre track analysis with genre that have more than 20 songs
SELECT GenreId,
	   MAX(Milliseconds / 60000) AS longest_track,
	   MIN(Milliseconds / 60000) AS shortest_track,
	   AVG(Milliseconds / 60000) AS avg_track_length
FROM Track 
GROUP BY 1
HAVING COUNT(*) > 20

-- returns a revenue report per country
SELECT BillingCountry, 
	   Count(InvoiceId) AS num_invoice,
	   SUM(Total) AS total_invoice, 
	   AVG(Total) AS avg_invoice
FROM Invoice
GROUP BY 1
HAVING SUM(Total) > 50
ORDER BY total_invoice DESC 

--returns a count for each length range
SELECT CASE WHEN Milliseconds < 180000 THEN 'Short'
			WHEN Milliseconds >= 180000 AND Milliseconds <= 300000 THEN 'Medium'
			ELSE 'Long' END AS track_length,
		COUNT(*) AS count
FROM Track 
GROUP BY 1


--Union with aggregation
SELECT c.Country, SUM(i.Total)
FROM Customer c
LEFT JOIN Invoice i ON c.CustomerId = i.CustomerId 
GROUP BY 1
UNION ALL
SELECT 'Sum of Countries', SUM(i.Total)
FROM Customer c
LEFT JOIN Invoice i ON c.CustomerId = i.CustomerId 
ORDER BY 2 DESC 

--Union across joined tables
SELECT 'Customer', 
		c.FirstName, 
		c.LastName, SUM(i.Total) AS total
FROM Customer c
LEFT JOIN Invoice i ON c.CustomerId = i.CustomerId 
GROUP BY 1,2,3
UNION 
SELECT 'Employee',
		e.FirstName, e.LastName, SUM(i.Total) 
FROM Employee e
LEFT JOIN Customer c ON e.EmployeeId = c.SupportRepId 
LEFT JOIN Invoice i ON c.CustomerId = i.CustomerId 
GROUP BY 1,2,3
